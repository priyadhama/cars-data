def main(cnt,b):
    q1 = np.percentile(cnt[b],25)
    q2 = np.percentile(cnt[b],50)
    q3 = np.percentile(cnt[b],75)
    iqr = q3 - q1
    lower_fence = q1 - (iqr*1.5)
    upper_fence = q3 +  (iqr*1.5)
    lower_fence,upper_fence
    cnt[b] = np.where(((cnt[b]<lower_fence)|(cnt[b]>upper_fence)),q2,cnt[b])
for i in ['ID','price','model Year','mileage','no. of Cylinders','displacement','hp','weight','acceleration','origin']:
    main(cnt,i)
